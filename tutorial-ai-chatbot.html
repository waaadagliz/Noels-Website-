This tutorial teaches you how to build a basic chatbot using HTML, CSS, and JavaScript. It's a fun beginner project to understand input handling and response logic.

Step 1: HTML

<div id="chatbox"></div>
<input type="text" id="userInput" placeholder="Ask me something..." />
<button onclick="sendMessage()">Send</button>

Step 2: JavaScript

function sendMessage() {
  const input = document.getElementById("userInput").value;
  const chatbox = document.getElementById("chatbox");
  const lower = input.toLowerCase();
  let response = "";

  if (lower.includes("hello")) {
    response = "Hi there! üëã";
  } else if (lower.includes("how are you")) {
    response = "I'm just code, but I'm doing great!";
  } else {
    response = "Sorry, I don‚Äôt understand that.";
  }

  chatbox.innerHTML += `<p><strong>You:</strong> ${input}</p>`;

  let botParagraph = document.createElement("p");
  botParagraph.innerHTML = `<strong>Bot:</strong> `;
  chatbox.appendChild(botParagraph);

  let i = 0;
  let typing = setInterval(() => {
    if (i < response.length) {
      botParagraph.innerHTML += response.charAt(i);
      i++;
    } else {
      clearInterval(typing);
    }
  }, 50);

  document.getElementById("userInput").value = "";
}
